Function implementation

function Ie = expand2d_base(I, filter, order)

dealing with multispectral images

[X,Y,C] = size(I);

if C>1
    Ie = zeros(X*2,Y*2,C);
    for c=1:C
        Ie(:,:,c) = expand2d_base(I(:,:,c), filter, order);
    end
    return;
end

main program

if strcmpi(filter,'lpl')

    h = [1 4 6 4 1]/16;

else

end

end % end of expand2d_base
