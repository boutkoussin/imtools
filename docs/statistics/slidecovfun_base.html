    function wc = covint(P, Q, X, x, y, ws)
        xp = x + (2*ws+1);  yp = y + (2*ws+1);
        xpyp = (yp-1)*X+xp;  xy = (y-1)*X+x;
        xpy = (y-1)*X+xp;  xyp = (yp-1)*X+x;
        n = (xp-x) * (yp-y);
        p = P{xpyp} + P{xy} - P{xyp} - P{xpy};
        q = Q{xy} + Q{xpyp} - Q{xpy} - Q{xyp};
        wc = 1/(n-1) * (q - 1/n * transpose(p) * p);
    end
